Overview

This class is responsible for all the interactions with Google Sheets API. The main purpose of this class is to contain all the operations needed to read/update/delete information to/from Google sheets.

def self.read_cell_values(spreadsheet_id, range)
  OHANA.logger.debug("Reading sheet ID: '#{spreadsheet_id}', range: #{range}")
  response = OHANA.ss_service.get_spreadsheet_values(spreadsheet_id, range)
  response.values
end

def self.write_cell_values(spreadsheet_id, range, values)
  value_range_object = Google::Apis::SheetsV4::ValueRange.new(range: range, values: values)
  OHANA.ss_service.update_spreadsheet_value(spreadsheet_id, range, value_range_object,
                                            value_input_option: 'RAW')
end
