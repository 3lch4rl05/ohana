def run
  args = Hash[ARGV.flat_map { |s| s.scan(/--?([^=\s]+)(?:=(\S+))?/) }]
  if args.key?('version')
    ver = Gem.loaded_specs['ohana'].version
    puts "Ohana #{ver}"
    exit
  end

  OHANA.props = YAML.load_file('app_props.yml')
  ohana = OHANA::OhanaInit.new
  ohana.init(args)

  at_exit do
    puts "\nFinishing execution... :)\n\n".italic
  end
end
